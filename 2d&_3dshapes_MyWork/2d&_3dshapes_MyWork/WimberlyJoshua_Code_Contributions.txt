**Area Calculation Methods**

public static double rectangleArea(double length, double width) {
    return length * width;
}
public static double circleArea(double radius) {
    return Math.PI * radius * radius;
}
public static double squareArea(double width) {
    return width * width;
}
public static double triangleArea(double base, double height) {
    return 0.5 * base * height;
}
public static double parallelogramArea(double base, double height) {
    return base * height;
}

private void drawCircle() {
    try {
        String input = JOptionPane.showInputDialog(this, "Enter radius:");
        if (input == null) return;
        double radius = Double.parseDouble(input);
        if (radius <= 0) {
            JOptionPane.showMessageDialog(this, "Radius must be positive!");
            return;
        }
        double area = Math.PI * radius * radius;
        JOptionPane.showMessageDialog(this, "Area = " + String.format("%.2f", area));
        drawPanel.setCircle(radius);
    } catch (NumberFormatException e) {
        JOptionPane.showMessageDialog(this, "Invalid number!");
    }
}

**Custom Drawing Panel**

class DrawPanel extends JPanel {
    private String shapeType = "";
    private double[] dimensions = {};
    
    protected void paintComponent(Graphics g) {
        super.paintComponent(g);
        Graphics2D g2d = (Graphics2D) g;
        g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
        
        // Drawing reference grid lines
        g2d.setColor(Color.LIGHT_GRAY);
        g2d.drawLine(centerX, 0, centerX, getHeight());
        g2d.drawLine(0, centerY, getWidth(), centerY);
        
        // Shape-specific drawing with scaling
        if (shapeType.equals("circle") && dimensions.length > 0) {
            int radius = (int) (dimensions[0] * 20);
            g2d.drawOval(centerX - radius, centerY - radius, radius * 2, radius * 2);
            g2d.drawString("Circle (r=" + dimensions[0] + ")", centerX - 50, centerY + radius + 20);
        }
    }
}

**Interactive Mouse Controls**

addMouseListener(new MouseAdapter() {
    @Override
    public void mousePressed(MouseEvent e) {
        isDragging = true;
        lastMouseX = e.getX();
        lastMouseY = e.getY();
    }
    
    @Override
    public void mouseReleased(MouseEvent e) {
        isDragging = false;
    }
});

addMouseMotionListener(new MouseMotionAdapter() {
    @Override
    public void mouseDragged(MouseEvent e) {
        if (isDragging) {
            int deltaX = e.getX() - lastMouseX;
            int deltaY = e.getY() - lastMouseY;
            shapeX += deltaX;
            shapeY += deltaY;
            lastMouseX = e.getX();
            lastMouseY = e.getY();
            repaint();
        }
    }
});

private void drawThreeTriangles(Graphics2D g2d) {
    int centerX = shapeX;
    int centerY = shapeY;
    int triangleSize = 80;
    
    // Triangle 1 - Red (top left)
    g2d.setColor(Color.RED);
    int[] x1 = {centerX - triangleSize, centerX - triangleSize/2, centerX};
    int[] y1 = {centerY - triangleSize/2, centerY - triangleSize, centerY - triangleSize/2};
    g2d.fillPolygon(x1, y1, 3);

    // Triangle 2 - Blue (top right)
    g2d.setColor(Color.BLUE);
    int[] x2 = {centerX, centerX + triangleSize/2, centerX + triangleSize};
    int[] y2 = {centerY - triangleSize/2, centerY - triangleSize, centerY - triangleSize/2};
    g2d.fillPolygon(x2, y2, 3);

    // Triangle 3 - Green (bottom center)
    g2d.setColor(Color.GREEN);
    int[] x3 = {centerX - triangleSize/2, centerX, centerX + triangleSize/2};
    int[] y3 = {centerY + triangleSize/2, centerY + triangleSize, centerY + triangleSize/2};
    g2d.fillPolygon(x3, y3, 3);
}
